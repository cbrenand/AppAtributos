from flask import Flask, redirect, request, jsonify
import requests

app = Flask(__name__)

# Credenciais do Mercado Livre
CLIENT_ID = "SEU_APP_ID"
CLIENT_SECRET = "SEU_CLIENT_SECRET"
REDIRECT_URI = "http://127.0.0.1:5000/auth"  # Altere se necessário

@app.route('/login')
def login():
    return redirect(f"https://auth.mercadolivre.com.br/authorization?response_type=code&client_id={CLIENT_ID}&redirect_uri={REDIRECT_URI}")

@app.route('/auth')
def auth():
    code = request.args.get('code')
    if not code:
        return jsonify({"error": "Nenhum código de autorização encontrado."})

    # Trocar o código por um token
    url = "https://api.mercadolibre.com/oauth/token"
    payload = {
        "grant_type": "authorization_code",
        "client_id": CLIENT_ID,
        "client_secret": CLIENT_SECRET,
        "code": code,
        "redirect_uri": REDIRECT_URI
    }
    headers = {"Content-Type": "application/x-www-form-urlencoded"}
    response = requests.post(url, data=payload, headers=headers)

    if response.status_code == 200:
        return jsonify(response.json())
    else:
        return jsonify({"error": "Erro ao trocar o código pelo token", "details": response.json()})

if __name__ == '__main__':
    app.run(debug=True)
